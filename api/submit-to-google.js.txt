// Ping Google about each page
function pingGoogle(url) {
  return new Promise((resolve) => {
    const googleUrl = `https://www.google.com/ping?sitemap=${encodeURIComponent(url)}`;
    
    https.get(googleUrl, (res) => {
      console.log(`✅ Pinged: ${url} - Status: ${res.statusCode}`);
      resolve();
    }).on('error', (err) => {
      console.log(`❌ Failed: ${url} - Error: ${err.message}`);
      resolve();
    });
  });
}

// Also ping Bing
function pingBing(url) {
  return new Promise((resolve) => {
    const bingUrl = `https://www.bing.com/ping?sitemap=${encodeURIComponent(url)}`;
    
    https.get(bingUrl, (res) => {
      console.log(`✅ Bing pinged: ${url} - Status: ${res.statusCode}`);
      resolve();
    }).on('error', (err) => {
      console.log(`❌ Bing failed: ${url} - Error: ${err.message}`);
      resolve();
    });
  });
}

// Run for all pages
async function submitAll() {
  console.log('🚀 Submitting all pages to Google and Bing...\n');
  
  for (const page of pages) {
    await pingGoogle(page);
    await new Promise(resolve => setTimeout(resolve, 500)); // Wait 0.5 seconds between pings
    await pingBing(page);
    await new Promise(resolve => setTimeout(resolve, 500));
  }
  
  console.log('\n✅ All pages submitted!');
}

submitAll();
