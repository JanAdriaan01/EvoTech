import { Resend } from 'resend'; const resend = new Resend(process.env.RESEND_API_KEY); export default async function handler(req, res) { if (req.method !== 'POST') { return res.status(405).json({ error: 'Method not allowed' }); } try { const data = req.body; // Validate required fields const required = ['customerName', 'customerPhone', 'propertyType', 'propertySize', 'location']; for (const field of required) { if (!data[field]) { return res.status(400).json({ error: `Missing required field: ${field}` }); } } // Send to admin const adminEmailContent = generateAdminEmail(data); await resend.emails.send({ from: 'Net Cam SA ', // Replace with your verified domain later to: ['info@netcamsa.co.za'], subject: `New Quote Request from ${data.customerName}`, html: adminEmailContent, }); // Send confirmation to customer if email provided if (data.customerEmail) { const customerEmailContent = generateCustomerEmail(data); await resend.emails.send({ from: 'Net Cam SA ', to: [data.customerEmail], subject: 'We received your request - Net Cam SA', html: customerEmailContent, }); } return res.status(200).json({ success: true }); } catch (error) { console.error('Resend error:', error); return res.status(500).json({ error: 'Failed to send email' }); } } function generateAdminEmail(data) { // Build detailed admin email (same as earlier) return `

New Quote Request

Name: ${data.customerName}

Phone: ${data.customerPhone}

Email: ${data.customerEmail || 'Not provided'}

Property: ${data.propertyType} - ${data.propertySize}

Location: ${data.location}

Categories: ${data.categories.join(', ')}

Full Details:

${JSON.stringify(data, null, 2)}
`; } function generateCustomerEmail(data) { return `

Thank you, ${data.customerName}!

We have received your requirements and will get back to you within 1 hour (7am-5pm weekdays).

You can also reach us directly:

📞 075 461 3153
💬 https://wa.me/27754613153

Best regards,
Net Cam SA Team

`; }
